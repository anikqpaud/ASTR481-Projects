bias_means = []
all_biases = np.concatenate((day1_biases, day2_biases, day3_biases))

start_date = dt.strptime(all_biases[0][0].header['DATE-OBS'], "%Y-%m-%dT%H:%M:%S.%f")
for i in range(len(all_biases)):
    relative_time = (dt.strptime(all_biases[i][0].header["DATE-OBS"], "%Y-%m-%dT%H:%M:%S.%f") - start_date).total_seconds() / 3600
    bias_dates.append(relative_time)
    bias_means.append(np.mean(all_biases[i][0].data))

print(f'Average Bias: {np.mean(bias_means):.1f} +/- {np.std(bias_means):.1f}')

plt.xlabel("Hours since first bias")
plt.ylabel("Mean count in bias")
plt.title("Changes in Bias Over Time")
